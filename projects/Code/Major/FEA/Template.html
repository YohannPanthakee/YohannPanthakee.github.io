Using the logic from the table show above, thhe previous function was updated to incoporate the logic and this is shown below. 

def function(SID,ts):
        A = sum(SID)
        B = A/20
        T1 = (np.mean(SID)/10)
        T2 = T1/10
        Phi_1 = np.radians(sum(SID))
        Phi_2 = np.radians((np.pi)/4)
        time_step = T2/ts
        times = np.arange(0,3*T1,time_step)

        y1 = []
        y2 = []
        y3 = []
        new = []

        for i, time in enumerate(times):
            y1.append( A * np.sin((((2*np.pi)/T1)*time)-Phi_1) )
            y2.append( B * np.cos((((2*np.pi)/T2)*time)-Phi_2) )

        for (a,b) in zip(y1,y2):
            y3.append( a + 2 * b )

        for values in y3:

            if values >= max(y3)/2:
                values = np.max(y3)
                new.append(values)

            elif np.absolute(values) < np.max(y3)/2:
                values = 0
                new.append(values)

            elif values <= np.min(y3)/2:
                values = np.min(y3)
                new.append(values)
